             <h5 class="modal-title">Modal title</h5>
             <button class="close" data-dismiss="modal"><span aria-hidden="true">&times;</span></button>
          </div>
          <div class="modal-body">
             <p>Woohoo, you're reading this text in a modal!</p>
          </div>

          </div>

    </div>
